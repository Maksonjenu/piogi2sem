    Lection 4.2

    Стили в XAML

    Стили в XAML позволяют определять и повторно использовать оформление элементов интерфейса. Это помогает поддерживать единообразный внешний вид приложения и упрощает внесение изменений.

    Основы работы со стилями

    Стиль в XAML определяется с помощью элемента Style и содержит набор Setter-ов, которые устанавливают значения свойств:

    <Style TargetType="Button">
    <Setter Property="Background" Value="Blue"/>
    <Setter Property="Foreground" Value="White"/>
    <Setter Property="FontSize" Value="14"/>
    <Setter Property="Padding" Value="10,5"/>
</Style>

    Применение стилей

    Стили можно применять несколькими способами:

    1. Неявно - через TargetType:
    <Window.Resources>
    <Style TargetType="Button">
        <Setter Property="Background" Value="Blue"/>
        <Setter Property="Foreground" Value="White"/>
    </Style>
</Window.Resources>

<!-- Стиль применится автоматически ко всем кнопкам -->
<Button Content="Кнопка с автоматическим стилем"/>
    1. Явно - через x:Key и свойство Style:
    <Window.Resources>
    <Style x:Key="PrimaryButton" TargetType="Button">
        <Setter Property="Background" Value="Blue"/>
        <Setter Property="Foreground" Value="White"/>
    </Style>
</Window.Resources>

<!-- Стиль применяется только к этой кнопке -->
<Button Content="Основная кнопка" Style="{StaticResource PrimaryButton}"/>

    Наследование стилей

    Стили могут наследоваться с помощью свойства BasedOn:

    <Window.Resources>
    <Style x:Key="BaseButton" TargetType="Button">
        <Setter Property="FontSize" Value="14"/>
        <Setter Property="Padding" Value="10,5"/>
    </Style>

    <Style x:Key="PrimaryButton" TargetType="Button" BasedOn="{StaticResource BaseButton}">
        <Setter Property="Background" Value="Blue"/>
        <Setter Property="Foreground" Value="White"/>
    </Style>

    <Style x:Key="SecondaryButton" TargetType="Button" BasedOn="{StaticResource BaseButton}">
        <Setter Property="Background" Value="Gray"/>
        <Setter Property="Foreground" Value="Black"/>
    </Style>
</Window.Resources>

    Конверторы данных

    Конверторы данных позволяют преобразовывать значения при привязке данных. Они реализуют интерфейс IValueConverter.

    Создание конвертера

    public class BooleanToVisibilityConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        bool boolValue = (bool)value;
        return boolValue ? Visibility.Visible : Visibility.Collapsed;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        Visibility visibility = (Visibility)value;
        return visibility == Visibility.Visible;
    }
}

    Использование конвертера в XAML

    <Window.Resources>
    <local:BooleanToVisibilityConverter x:Key="BoolToVis"/>
</Window.Resources>

<StackPanel>
    <CheckBox x:Name="ShowPanelCheckBox" Content="Показать панель"/>

    <StackPanel Visibility="{Binding IsChecked, ElementName=ShowPanelCheckBox, 
                           Converter={StaticResource BoolToVis}}">
        <TextBlock Text="Эта панель отображается только при включенном флажке"/>
    </StackPanel>
</StackPanel>

    Примеры полезных конверторов

    1. Конвертер для инвертирования логического значения:
    public class InverseBooleanConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        return !(bool)value;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        return !(bool)value;
    }
}
    1. Конвертер для преобразования строки в верхний регистр:
    public class StringToUpperConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        if (value is string str)
            return str.ToUpper();
        return value;
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        return value; // обычно не нужно конвертировать обратно
    }
}

    Триггеры

    Триггеры позволяют динамически изменять внешний вид и поведение элементов в ответ на события, изменения свойств или данных.

    Типы триггеров

    1. Property Triggers - реагируют на изменения значений свойств
    2. Data Triggers - реагируют на значения привязанных данных
    3. Event Triggers - реагируют на события
    4. MultiTriggers - комбинируют несколько условий

    Property Trigger (Триггер свойства)

    <Style TargetType="Button">
    <Setter Property="Background" Value="Blue"/>
    <Setter Property="Foreground" Value="White"/>

    <Style.Triggers>
        <!-- Срабатывает при наведении мыши (IsMouseOver = true) -->
        <Trigger Property="IsMouseOver" Value="True">
            <Setter Property="Background" Value="DarkBlue"/>
            <Setter Property="FontWeight" Value="Bold"/>
        </Trigger>

        <!-- Срабатывает при нажатии (IsPressed = true) -->
        <Trigger Property="IsPressed" Value="True">
            <Setter Property="Background" Value="Navy"/>
        </Trigger>

        <!-- Срабатывает когда кнопка отключена -->
        <Trigger Property="IsEnabled" Value="False">
            <Setter Property="Opacity" Value="0.5"/>
        </Trigger>
    </Style.Triggers>
</Style>

    Data Trigger (Триггер данных)

    <Style TargetType="TextBlock">
    <Style.Triggers>
        <DataTrigger Binding="{Binding Status}" Value="Error">
            <Setter Property="Foreground" Value="Red"/>
            <Setter Property="FontWeight" Value="Bold"/>
        </DataTrigger>

        <DataTrigger Binding="{Binding Status}" Value="Warning">
            <Setter Property="Foreground" Value="Orange"/>
        </DataTrigger>

        <DataTrigger Binding="{Binding Status}" Value="Success">
            <Setter Property="Foreground" Value="Green"/>
        </DataTrigger>
    </Style.Triggers>
</Style>

    MultiTrigger (Мульти-триггер)

    <Style TargetType="Button">
    <Style.Triggers>
        <MultiTrigger>
            <MultiTrigger.Conditions>
                <Condition Property="IsMouseOver" Value="True"/>
                <Condition Property="IsEnabled" Value="True"/>
            </MultiTrigger.Conditions>
            <MultiTrigger.Setters>
                <Setter Property="Background" Value="Gold"/>
                <Setter Property="Foreground" Value="Black"/>
            </MultiTrigger.Setters>
        </MultiTrigger>
    </Style.Triggers>
</Style>

    Event Trigger (Триггер событий)

    <Button Content="Анимированная кнопка">
    <Button.Triggers>
        <EventTrigger RoutedEvent="Button.MouseEnter">
            <BeginStoryboard>
                <Storyboard>
                    <DoubleAnimation 
                        Storyboard.TargetProperty="Width"
                        From="100" To="150" Duration="0:0:0.3"/>
                </Storyboard>
            </BeginStoryboard>
        </EventTrigger>

        <EventTrigger RoutedEvent="Button.MouseLeave">
            <BeginStoryboard>
                <Storyboard>
                    <DoubleAnimation 
                        Storyboard.TargetProperty="Width"
                        To="100" Duration="0:0:0.3"/>
                </Storyboard>
            </BeginStoryboard>
        </EventTrigger>
    </Button.Triggers>
</Button>

    Практические примеры

    Пример 1: Создание собственной темы для приложения

    <Application.Resources>
    <!-- Цветовая палитра -->
    <Color x:Key="PrimaryColor">#3498db</Color>
    <Color x:Key="SecondaryColor">#2ecc71</Color>
    <Color x:Key="AccentColor">#e74c3c</Color>
    <Color x:Key="BackgroundColor">#ecf0f1</Color>
    <Color x:Key="TextColor">#2c3e50</Color>

    <!-- Кисти на основе цветов -->
    <SolidColorBrush x:Key="PrimaryBrush" Color="{StaticResource PrimaryColor}"/>
    <SolidColorBrush x:Key="SecondaryBrush" Color="{StaticResource SecondaryColor}"/>
    <SolidColorBrush x:Key="AccentBrush" Color="{StaticResource AccentColor}"/>
    <SolidColorBrush x:Key="BackgroundBrush" Color="{StaticResource BackgroundColor}"/>
    <SolidColorBrush x:Key="TextBrush" Color="{StaticResource TextColor}"/>

    <!-- Базовый стиль для кнопок -->
    <Style TargetType="Button">
        <Setter Property="Background" Value="{StaticResource PrimaryBrush}"/>
        <Setter Property="Foreground" Value="White"/>
        <Setter Property="Padding" Value="15,5"/>
        <Setter Property="BorderThickness" Value="0"/>
        <Setter Property="FontSize" Value="14"/>
        <Setter Property="Cursor" Value="Hand"/>
        <Setter Property="Template">
            <Setter.Value>
                <ControlTemplate TargetType="Button">
                    <Border Background="{TemplateBinding Background}" 
                            CornerRadius="4"
                            Padding="{TemplateBinding Padding}">
                        <ContentPresenter HorizontalAlignment="Center" 
                                          VerticalAlignment="Center"/>
                    </Border>
                </ControlTemplate>
            </Setter.Value>
        </Setter>
        <Style.Triggers>
            <Trigger Property="IsMouseOver" Value="True">
                <Setter Property="Background">
                    <Setter.Value>
                        <SolidColorBrush Color="#2980b9"/>
                    </Setter.Value>
                </Setter>
            </Trigger>
            <Trigger Property="IsPressed" Value="True">
                <Setter Property="Background">
                    <Setter.Value>
                        <SolidColorBrush Color="#1a5276"/>
                    </Setter.Value>
                </Setter>
            </Trigger>
            <Trigger Property="IsEnabled" Value="False">
                <Setter Property="Opacity" Value="0.5"/>
            </Trigger>
        </Style.Triggers>
    </Style>

    <!-- Другие стили для разных элементов... -->
</Application.Resources>

    Пример 2: Конвертер для отображения статуса задачи

    public class TaskStatusToColorConverter : IValueConverter
{
    public object Convert(object value, Type targetType, object parameter, CultureInfo culture)
    {
        if (value is string status)
        {
            switch (status.ToLower())
            {
                case "completed":
                    return new SolidColorBrush(Colors.Green);
                case "in progress":
                    return new SolidColorBrush(Colors.Blue);
                case "not started":
                    return new SolidColorBrush(Colors.Gray);
                case "overdue":
                    return new SolidColorBrush(Colors.Red);
                default:
                    return new SolidColorBrush(Colors.Black);
            }
        }
        return new SolidColorBrush(Colors.Black);
    }

    public object ConvertBack(object value, Type targetType, object parameter, CultureInfo culture)
    {
        return DependencyProperty.UnsetValue;
    }
}

    Использование:

    <Window.Resources>
    <local:TaskStatusToColorConverter x:Key="StatusToColor"/>
</Window.Resources>

<ListView ItemsSource="{Binding Tasks}">
    <ListView.ItemTemplate>
        <DataTemplate>
            <StackPanel Orientation="Horizontal">
                <TextBlock Text="{Binding Name}" FontWeight="Bold"/>
                <TextBlock Text="{Binding Status}" 
                           Foreground="{Binding Status, Converter={StaticResource StatusToColor}}"
                           Margin="10,0,0,0"/>
            </StackPanel>
        </DataTemplate>
    </ListView.ItemTemplate>
</ListView>

                        </BeginStoryboard>
                    </EventTrigger.Actions>
                </EventTrigger>
                <EventTrigger RoutedEvent="Button.MouseLeave">
                    <EventTrigger.Actions>
                        <BeginStoryboard>
                            <Storyboard>
                                <ColorAnimation 
                                    Storyboard.TargetProperty="(Button.Background).(SolidColorBrush.Color)" 
                                    To="#3498db" 
                                    Duration="0:0:0.2"/>
                                <DoubleAnimation 
                                    Storyboard.TargetProperty="Width" 
                                    To="150" 
                                    Duration="0:0:0.2"/>
                            </Storyboard>

    Заключение

    • Стили позволяют унифицировать внешний вид элементов и сделать его легко изменяемым.
    • Конверторы обеспечивают трансформацию данных между источником и целевым элементом.
    • Триггеры отвечают за динамическое изменение свойств элементов в ответ на различные события и условия.

    Совместное использование этих инструментов позволяет создавать современные, интерактивные и эстетически привлекательные интерфейсы без необходимости писать большое количество кода на C#.
